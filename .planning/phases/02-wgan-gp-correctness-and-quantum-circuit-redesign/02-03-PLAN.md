Replace the per-sample loop with parameter broadcasting:
```python
for t in range(self.n_critic):
    self.c_optimizer.zero_grad()

    # Real batch: (batch_size, window_length) -> (batch_size, 1, window_length)
    real_batch_tensor = real_batch.unsqueeze(1).double()
    actual_batch_size = real_batch.shape[0]

    # Generate fake batch using parameter broadcasting (PERF-05)
    # Noise shape: (num_qubits, actual_batch_size) for correct broadcasting
    noise = torch.tensor(
        np.random.uniform(0, 4 * np.pi, size=(self.num_qubits, actual_batch_size)),
        dtype=torch.float32
    )
    results = self.generator(noise, self.params_pqc)
    gen_output = torch.stack(list(results)).T  # (batch_size, WINDOW_LENGTH)
    gen_output = gen_output.to(torch.float64) * GEN_SCALE  # BUG-02: consistent scaling

    # Reshape for Conv1D: (batch_size, 1, window_length)
    fake_batch_tensor = gen_output.unsqueeze(1)

    # Critic scores
    real_scores = self.critic(real_batch_tensor)
    fake_scores = self.critic(fake_batch_tensor.detach())  # Detach from generator graph

    # Wasserstein loss
    real_score_mean = torch.mean(real_scores)
    fake_score_mean = torch.mean(fake_scores)

    # One-sided gradient penalty with per-sample alpha
    alpha = torch.rand(actual_batch_size, 1, 1, device=real_batch_tensor.device, dtype=torch.float64)
    interpolated = alpha * real_batch_tensor + (1 - alpha) * fake_batch_tensor.detach()
    interpolated.requires_grad_(True)

    interpolated_scores = self.critic(interpolated)
    gradients = torch.autograd.grad(
        outputs=interpolated_scores,
        inputs=interpolated,
        grad_outputs=torch.ones_like(interpolated_scores),
        create_graph=True,
        retain_graph=True,
        only_inputs=True
    )[0]

    grad_norms = gradients.norm(2, dim=[1, 2])
    # One-sided: only penalize when gradient norm > 1
    gradient_penalty = torch.mean(torch.clamp(grad_norms - 1, min=0) ** 2)

    critic_loss = fake_score_mean - real_score_mean + self.lambda_gp * gradient_penalty
    critic_loss.backward()
    self.c_optimizer.step()

    critic_t_sum += critic_loss.item()
```

IMPORTANT: Use `actual_batch_size = real_batch.shape[0]` (not self.batch_size) for alpha shape to handle variable last batch. Use `fake_batch_tensor.detach()` in critic scores and interpolation to avoid computing generator gradients during critic training. Use `.item()` when accumulating critic_t_sum.

```python
def stylized_facts(self, original_data, fake_original, gen_windows=None):
    """Compute stylized facts RMSEs and kurtosis difference.

    Computes at stitched time series level. Also reports window-level if gen_windows provided.
    Includes: ACF (absence of autocorrelation), volatility clustering (ACF of |r|),
    leverage effect (corr of r_t and |r_{t+lag}|^2), heavy tails (kurtosis).
    """
    # Convert to numpy
    fake_np = fake_original.detach().cpu().numpy() if isinstance(fake_original, torch.Tensor) else np.asarray(fake_original)
    orig_np = original_data.detach().cpu().numpy() if isinstance(original_data, torch.Tensor) else np.asarray(original_data)

    # --- Stitched level (1D series) ---
    # ACF
    acf_gen = sm.tsa.acf(fake_np, nlags=18)[1:]
    acf_orig = sm.tsa.acf(orig_np, nlags=18)[1:]
    rmse_acf = np.sqrt(np.mean((acf_orig - acf_gen) ** 2))

    # Volatility clustering (ACF of absolute values)
    vol_gen = sm.tsa.acf(np.abs(fake_np), nlags=18)[1:]
    vol_orig = sm.tsa.acf(np.abs(orig_np), nlags=18)[1:]
    rmse_vol = np.sqrt(np.mean((vol_orig - vol_gen) ** 2))

    # Leverage effect
    lev_gen = []
    lev_orig = []
    for lag in range(1, 19):
        # Generated
        r_t = fake_np[:-lag]
        sq_lag = np.square(np.abs(fake_np[lag:]))
        lev_gen.append(np.corrcoef(r_t, sq_lag)[0, 1])
        # Original
        r_t_o = orig_np[:-lag]
        sq_lag_o = np.square(np.abs(orig_np[lag:]))
        lev_orig.append(np.corrcoef(r_t_o, sq_lag_o)[0, 1])
    rmse_lev = np.sqrt(np.mean((np.array(lev_orig) - np.array(lev_gen)) ** 2))

    # Heavy tails: excess kurtosis difference (WGAN-08)
    from scipy.stats import kurtosis as scipy_kurtosis
    kurt_orig = scipy_kurtosis(orig_np, fisher=True)  # excess kurtosis
    kurt_gen = scipy_kurtosis(fake_np, fisher=True)
    kurt_diff = abs(kurt_orig - kurt_gen)

    # --- Window-level (if windows provided) ---
    if gen_windows is not None:
        # Compute per-window stats and average
        # This is informational — the RMSEs above are the primary metrics
        pass  # Window-level stats printed but not returned as primary metric

    # EMD removed from here — computed in eval block on normalized space
    return rmse_acf, rmse_vol, rmse_lev, kurt_diff
```
