**2. Rewrite define_generator_circuit():**
Implement the data re-uploading pattern. The complete new circuit:

```python
def define_generator_circuit(self, noise_params, params_pqc):
    idx = 0

    # Step 1: Hadamard initialization for superposition
    for qubit in range(self.num_qubits):
        qml.Hadamard(wires=qubit)

    # Step 2: Initial noise encoding (RX for non-commutativity with RZ variational)
    self.encoding_layer(noise_params)

    # Step 3: Variational layers with data re-uploading
    for layer in range(self.num_layers):
        # Trainable rotations: Rot(phi, theta, omega) per qubit
        for qubit in range(self.num_qubits):
            qml.Rot(phi=params_pqc[idx],
                     theta=params_pqc[idx + 1],
                     omega=params_pqc[idx + 2],
                     wires=qubit)
            idx += 3

        # Range-based entangling CNOTs (PennyLane strongly entangling pattern)
        if self.num_qubits > 1:
            range_param = (layer % (self.num_qubits - 1)) + 1
            for qubit in range(self.num_qubits):
                target_qubit = (qubit + range_param) % self.num_qubits
                qml.CNOT(wires=[qubit, target_qubit])

        # Data re-uploading: re-encode noise after each variational layer
        self.encoding_layer(noise_params)

    # Step 4: Final measurement preparation rotations
    for qubit in range(self.num_qubits):
        qml.RX(phi=params_pqc[idx], wires=qubit)
        idx += 1
        qml.RY(phi=params_pqc[idx], wires=qubit)
        idx += 1

    # Step 5: PauliX + PauliZ measurements (output dim = 2 * NUM_QUBITS)
    measurements = []
    for i in range(self.num_qubits):
        measurements.append(qml.expval(qml.PauliX(i)))
        measurements.append(qml.expval(qml.PauliZ(i)))

    return (*measurements,)
```

- PRESERVED: PauliX + PauliZ measurements (QC-03)
- PRESERVED: Range-based CNOT pattern
- PRESERVED: Final RX, RY measurement prep
