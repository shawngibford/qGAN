  for q in range(num_qubits): qml.Hadamard(wires=q)
  # Step 2: IQP RZ encoding with params_pqc[idx] for each qubit
  for q in range(num_qubits):

  # Step 4: strongly entangled layers

      qml.Rot(phi=params_pqc[idx], theta=params_pqc[idx+1], omega=params_pqc[idx+2], wires=q); idx += 3
    # Range-based CNOT pattern: r = (layer % (num_qubits - 1)) + 1
    r = (layer % (num_qubits - 1)) + 1
    for q in range(num_qubits):

      qml.CNOT(wires=[q, target])
  # Step 5: Final RX/RY measurement prep
  for q in range(num_qubits): qml.RX(phi=params_pqc[idx], wires=q); idx += 1
  for q in range(num_qubits): qml.RY(phi=params_pqc[idx], wires=q); idx += 1
  return tuple(qml.expval(qml.PauliX(i)) for i in range(num_qubits)) + tuple(qml.expval(qml.PauliZ(i)) for i in range(num_qubits))
