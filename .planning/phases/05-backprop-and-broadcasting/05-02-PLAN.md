# -- Phase 5 Validation: Equivalence Test (ONE-TIME -- remove after confirmation) --
# Verifies batched QNode output matches sequential per-sample output within 1e-6
import torch, numpy as np

torch.manual_seed(42)
np.random.seed(42)

TEST_BATCH = 12  # match training batch_size

# Generate test inputs
noise_seq = torch.tensor(
    np.random.uniform(0, 4 * np.pi, size=(NUM_QUBITS, TEST_BATCH)),
    dtype=torch.float32
)
par_light_seq = torch.rand(NUM_QUBITS, TEST_BATCH, dtype=torch.float32)  # (num_qubits, batch)

# Sequential: call QNode TEST_BATCH times one-by-one
with torch.no_grad():
    sequential_results = []
    for i in range(TEST_BATCH):
        out = qgan.generator(noise_seq[:, i], par_light_seq[:, i], qgan.params_pqc)
        sequential_results.append(torch.stack(list(out)))
    sequential_out = torch.stack(sequential_results)  # (TEST_BATCH, window_length)

# Batched: single call with shape (num_qubits, batch)
with torch.no_grad():
    results = qgan.generator(noise_seq, par_light_seq, qgan.params_pqc)
    batched_out = torch.stack(list(results)).T  # (TEST_BATCH, window_length)

# Compare
max_diff = (sequential_out - batched_out).abs().max().item()
passed = max_diff < 1e-6
print(f'Equivalence test: max_diff={max_diff:.2e}  {"PASS" if passed else "FAIL -- investigate before continuing"}')
assert passed, f'Equivalence test FAILED: max_diff={max_diff:.2e} exceeds 1e-6 tolerance'
print('Batched output matches sequential within 1e-6 tolerance')
```

# Runs 20 epochs twice with identical seed to confirm broadcasting produces reproducible training.
# Research verified backprop+broadcasting matches sequential gradients within 1e-17;
# this confirms the full training loop (data sampling, loss, optimizer) is deterministic.

MINI_EPOCHS = 20
SEED = 123

    torch.manual_seed(seed)
    np.random.seed(seed)
    model = qGAN(
        num_epochs=epochs,
        batch_size=BATCH_SIZE,
        window_length=WINDOW_LENGTH,
        n_critic=N_CRITIC,
        gp=LAMBDA,
        num_layers=NUM_LAYERS,
        num_qubits=NUM_QUBITS,

    g_opt = torch.optim.Adam([model.params_pqc], lr=LR_GENERATOR, betas=(0.0, 0.9))

    start = time.time()

# -- Phase 5 Validation: Timing / SC4 Gate (ONE-TIME -- remove after confirmation) --
# SC4: epoch wall-clock time must be < 30% of pre-broadcasting baseline
# Pre-broadcasting baseline: run 10 epochs using parameter-shift + sequential per-sample calls
# Post-broadcasting: run 10 epochs with current backprop + broadcasting implementation
import time, types
import pennylane as qml

TIMING_EPOCHS = 10
TIMING_SEED = 456

# --- Post-broadcasting timing (current implementation) ---
print(f'Measuring post-broadcasting timing ({TIMING_EPOCHS} epochs)...')
torch.manual_seed(TIMING_SEED)
np.random.seed(TIMING_SEED)
post_model = qGAN(
    num_epochs=TIMING_EPOCHS, batch_size=BATCH_SIZE, window_length=WINDOW_LENGTH,
    n_critic=N_CRITIC, gp=LAMBDA, num_layers=NUM_LAYERS, num_qubits=NUM_QUBITS, delta=delta,
)
c_post = torch.optim.Adam(post_model.critic.parameters(), lr=LR_CRITIC, betas=(0.0, 0.9))
g_post = torch.optim.Adam([post_model.params_pqc], lr=LR_GENERATOR, betas=(0.0, 0.9))
post_model.compile_QGAN(c_post, g_post)
post_start = time.time()
post_model.train_qgan(dataloader, log_delta, transformed_norm_log_delta, num_elements)
post_total = time.time() - post_start
post_per_epoch = post_total / TIMING_EPOCHS

# --- Pre-broadcasting timing (parameter-shift + sequential per-sample) ---
print(f'Measuring pre-broadcasting reference ({TIMING_EPOCHS} epochs, sequential per-sample)...')
torch.manual_seed(TIMING_SEED)
np.random.seed(TIMING_SEED)
pre_model = qGAN(
    num_epochs=TIMING_EPOCHS, batch_size=BATCH_SIZE, window_length=WINDOW_LENGTH,
    n_critic=N_CRITIC, gp=LAMBDA, num_layers=NUM_LAYERS, num_qubits=NUM_QUBITS, delta=delta,
)
# Override generator to use parameter-shift (sequential baseline)
pre_model.generator = qml.QNode(
    pre_model.define_generator_circuit,
    pre_model.quantum_dev,
    interface='torch',
    diff_method='parameter-shift'
)
c_pre = torch.optim.Adam(pre_model.critic.parameters(), lr=LR_CRITIC, betas=(0.0, 0.9))
g_pre = torch.optim.Adam([pre_model.params_pqc], lr=LR_GENERATOR, betas=(0.0, 0.9))
pre_model.compile_QGAN(c_pre, g_pre)

# Sequential epoch: one QNode call per sample (pre-broadcasting pattern)
def _sequential_epoch(self, gan_data_list, par_data_list, original_data, preprocessed_data, epoch):
    critic_t_sum = 0
    for t in range(self.n_critic):
        self.c_optimizer.zero_grad()
        real_batch, fake_batch = [], []
        for i in range(self.batch_size):
            idx = torch.randint(0, len(gan_data_list), (1,)).item()
            rl = gan_data_list[idx].reshape(1, self.window_length)
            pw = par_data_list[idx].reshape(1, self.window_length)
            real_batch.append(torch.stack([rl, pw], dim=1).double())
            with torch.no_grad():
                noise = torch.tensor(np.random.uniform(0, 4*np.pi, size=self.num_qubits), dtype=torch.float32)
                pc = pw.reshape(self.num_qubits, 2).mean(dim=1).float()
                pc = (pc + 1.0) / 2.0
                go = self.generator(noise, pc, self.params_pqc)
                if isinstance(go, (list, tuple)): go = torch.stack(list(go))
                go = go.to(torch.float64) * 0.1
                fake_batch.append(torch.stack([go.unsqueeze(0), pw.double()], dim=1))
        rt = torch.cat(real_batch, dim=0)
        ft = torch.cat(fake_batch, dim=0)
        rs = self.critic(rt); fs = self.critic(ft)
        alpha = torch.rand(self.batch_size, 1, 1)
        interp = (alpha * rt + (1-alpha) * ft).requires_grad_(True)
        gp = ((torch.autograd.grad(self.critic(interp), interp,
               torch.ones_like(self.critic(interp)), True, True)[0].norm(2, dim=[1,2]) - 1)**2).mean()
        loss = fs.mean() - rs.mean() + self.gp * gp
        loss.backward(); self.c_optimizer.step()
        critic_t_sum += loss
    self.critic_loss_avg.append(critic_t_sum / self.n_critic)
    # Generator step (sequential)
    inputs, par_cs, par_ws = [], [], []
    for _ in range(self.batch_size):
        idx = torch.randint(0, len(par_data_list), (1,)).item()
        pw = par_data_list[idx].reshape(self.window_length)
        par_ws.append(pw)
        pc = pw.reshape(self.num_qubits, 2).mean(dim=1).float()
        pc = (pc + 1.0) / 2.0
        par_cs.append(pc)
        inputs.append(torch.tensor(np.random.uniform(0, 4*np.pi, size=self.num_qubits), dtype=torch.float32))
    self.g_optimizer.zero_grad()
    gen_samps = []
    for inp, pc in zip(inputs, par_cs):
        go = self.generator(inp, pc, self.params_pqc)
        if isinstance(go, (list, tuple)): go = torch.stack(list(go))
        gen_samps.append(go.to(torch.float64) * 0.1)
    gs = torch.stack(gen_samps)
    pw_t = torch.stack(par_ws).double()
    gi = torch.stack([gs, pw_t], dim=1)
    gloss = -self.critic(gi).mean()
    gloss.backward(); self.g_optimizer.step()
    self.generator_loss_avg.append(gloss)

pre_model._train_one_epoch = types.MethodType(_sequential_epoch, pre_model)
pre_start = time.time()
pre_model.train_qgan(dataloader, log_delta, transformed_norm_log_delta, num_elements)
pre_total = time.time() - pre_start
pre_per_epoch = pre_total / TIMING_EPOCHS

# SC4 gate
ratio = post_per_epoch / pre_per_epoch
speedup = pre_per_epoch / post_per_epoch
sc4_pass = ratio < 0.30

print(f'\n-- Phase 5 Timing Results --')
print(f'Pre-broadcasting:  {pre_per_epoch:.2f}s/epoch  (total {pre_total:.1f}s for {TIMING_EPOCHS} epochs)')
print(f'Post-broadcasting: {post_per_epoch:.2f}s/epoch (total {post_total:.1f}s for {TIMING_EPOCHS} epochs)')
print(f'Speedup:           {speedup:.1f}x  (ratio: {ratio:.1%} of pre-broadcasting time)')
print(f'SC4 (<30%):        {"PASS" if sc4_pass else "FAIL -- ratio exceeds 30% threshold, investigate before Phase 6"}')

        f'Check: (1) diff_method is backprop not parameter-shift, (2) all 3 loops converted, '
        f'(3) no accidental sequential calls remaining.'
    )
```
